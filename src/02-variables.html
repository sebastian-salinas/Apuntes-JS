Variables

Inicialización y Reasignación

Las variables en JavaScript pueden ser inicializadas y reasignadas según sea necesario. A continuación, se muestran ejemplos: 

// Inicializar una variable con un valor
          
        var producto = "Monitor de 55 pulgadas";
        console.log(producto);
        
// Las variables se pueden reasignar

        producto = "Monitor de 60 pulgadas";
        console.log(producto);
        
// JavaScript es un lenguaje de tipado dinámico

        var producto = 55;
        console.log(producto);
        
// Declarar variables sin valor inicial

        var disponible;
        disponible = true;
        disponible = false;
        console.log(disponible);

Uso de Scope y Let

El scope de las variables var y let es fundamental para evitar problemas con nombres repetidos: 

// No se puede tener variables con el mismo nombre en el mismo scope

        var nombre = "Juan";
        var nombre = "Pedro";
        
// Se puede tener variables con el mismo nombre en diferentes scopes

        function mostrarNombre() {
          var nombre = "Juan";
          console.log(nombre);
        }
        
        mostrarNombre();
        
        function mostrarNombre() {
          var nombre = "Pedro";
          console.log(nombre);
        }
        
        mostrarNombre();
        
// Let permite iniciar variables y reasignar valores

        let precio;
        
        precio = 1450;
        
        console.log(precio);

Uso de Const

Las constantes en JavaScript deben ser inicializadas al declararse y no pueden ser reasignadas: 


// Constantes son variables que no se pueden reasignar ni tampoco se pueden iniciar sin asignar un valor

        const numeroPi = 3.14;
        
        console.log(numeroPi);
